CREATE TIMER tm1 INTERVAL '00:00:01' AS INSERT PROP1=1, t1=CURRENT_TIMESTAMP;
CREATE TIMER tm2 INTERVAL '00:00:10' AS INSERT PROP1=2, t2=CURRENT_TIMESTAMP;
CREATE TIMER tm3 INTERVAL '00:00:01' AS {${INSERT PROP1=3, t1=CURRENT_TIMESTAMP}, ${INSERT PROP1=4, t1=CURRENT_TIMESTAMP}};
SELECT * FROM afy:ClassOfTimers WHERE CONTAINS(afy:objectID, 'tm1');
SELECT * FROM afy:ClassOfTimers WHERE afy:objectID=.tm2;

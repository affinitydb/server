inheritance.sql

create class Person as select * where firstname is not null or lastname is not null;
create class Taxpayer as select * from Person where tax_id is not null;
create class Phd as select * from Person where title = 'Dr';

select * from Person;
select * from Phd;   
select * from Taxpayer;

select * from Taxpayer & Phd;
select * from Taxpayer;

select * from Person match against ('johannesburg');

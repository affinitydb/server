transaction_basic.sql

INSERT (a, b) VALUES (1, 2);
CREATE CLASS xacttest AS SELECT * WHERE  a IS NOT NULL AND b IS NOT NULL;
SELECT * FROM xacttest;

START TRANSACTION;
INSERT (a, b) VALUES (777, 777.777);
UPDATE xacttest SET b=3 WHERE a=1;
DELETE FROM xacttest WHERE a=1;
ROLLBACK;
SELECT * FROM xacttest;
